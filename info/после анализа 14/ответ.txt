class TelegramListener(QThread):
    update_received = Signal(dict)  # —Å–∏–≥–Ω–∞–ª –ø—Ä–∏ –ø–æ–ª—É—á–µ–Ω–∏–∏ callback

    def __init__(self, notifier):

        self.running = False
        self.offset = 0

    def run(self):
        self.running = True
        while self.running:
            url = f"https://api.telegram.org/bot{self.notifier.token}/getUpdates"
            params = {'timeout': 10, 'offset': self.offset}
            resp = requests.get(url, params=params, timeout=15).json()
            for update in resp.get('result', []):

                if 'callback_query' in update:

            time.sleep(1)
